159 void Controller::convertHsv(int& r, int& g, int& b)
160 {
161  // r,g,bの最大値を求める
162  double max = r;
163  if(max < g) max = g;
164  if(max < b) max = b;
165 
166  // r,g,bの最小値を求める
167  double min = r;
168  if(min > g) min = g;
169  if(min > b) min = b;
170 
171  // 色相(hue)を求める
172 
173  // 3つが同値の時は色相(hue)は０
174  if(r == g && r == b) hsv.hue = 0;
175 
176  // rが最大値の場合
177  else if(max == r) {
178  // 0除算を防ぐ処理
179  if(max - min != 0) max += 1;
180  hsv.hue = 60 * ((g - b) / (max - min));
181  }
182  // gが最大値の場合
183  else if(max == g) {
184  // 0除算を防ぐ処理
185  if(max - min != 0) max += 1;
186  hsv.hue = 60 * ((b - r) / (max - min)) + 120;
187  }
188  // bが最大値の場合
189  else if(max == b) {
190  // 0除算を防ぐ処理
191  if(max - min != 0) max += 1;
192  hsv.hue = 60 * ((r - g) / (max - min)) + 240;
193  }
194  //求められた色彩(hue)がマイナス値だった場合は360を加算して0～360の範囲に収める
195  if(hsv.hue < 0) hsv.hue += 360;
196 
197  // 0除算を防ぐ処理
198  if(max - min != 0) max += 1;
199 
200  // 彩度(saturation)を求める
201  hsv.saturation = (max - min) / max * 100;
202 
203  // 明度(value)を求める
204  hsv.value = max / 255 * 100;
205 }
